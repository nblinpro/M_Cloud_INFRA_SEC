Aide-mémoire Système & Réseau

Gestion des processus

# Voir les processus gourmands
htop

# Tuer un processus par son nom
pkill -f nom_processus

# Voir les ports ouverts
ss -tulpn

Gestion des Archives

# Compresser un dossier en tar.gz
tar -czvf archive.tar.gz dossier/

# Décompresser
tar -xzvf archive.tar.gz

Docker Quick Tips

# Nettoyer les ressources inutilisées
docker system prune -a

# Voir les logs en direct
docker logs -f id_conteneur

# Exécuter un shell dans un conteneur
docker exec -it id_conteneur /bin/bash
